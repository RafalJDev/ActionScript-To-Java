      private function getDigitLabel(meterNo:String, precisionNo:String):String
      {
        var no:Number = parseInt(meterNo);
        if (isNaN(no))
          no = 0;

        var precision:Number = parseInt(precisionNo);
        if (isNaN(precision))
          precision = 0;

        var digit:String = "";
        while (no-- > 0)
          digit += "0";

        if (precision > 0)
          digit += ",";

        while (precision-- > 0)
          digit += "0";

        return digit;
      }

      private function cbMeterKindChanged(event:Event):void
      {
        var oldView:View = m_currentView;
        m_currentView =  getView(cbMeterKind.value);
        if(m_currentView == null) //załadowanie
        {
          if (oldView)
            ncData.removeElement(oldView);

          loadView(cbMeterKind.value);
        }
        else if (oldView && oldView.className != m_currentView.className)
        {
          ncData.removeElement(oldView);
          ncData.addElement(m_currentView);
        }
        else if (!oldView)
          ncData.addElement(m_currentView);
      }

      private function getView(meterKind:String):*
      {
        var o:Object = m_viewList[meterKind];
        if(o == null)
          return null;

        return o["VIEW"];
      }

      private function loadView(meterKind:String):void
      {
        var o:Object = m_viewList[meterKind];
        if(o == null)
          return;

        var swf:String = o["SWF"];
        var func:String = o["FUNC"];


        var frm:SwfObject = new SwfObject(this.stage);
        frm.addResponder(new mx.rpc.Responder(createModalClassResult, showFaultEvent));
        frm.CreateObject(swf,func);
      }
      private function createModalClassResult(data:Object):void
      {
        m_currentView = UiCommon.getObject(data) as View;
        if (m_currentView == null)
          return;

        m_currentView.dbManager = grid.dbManager;
        ncData.addElement(m_currentView);
        var o:Object = m_viewList[cbMeterKind.value];
        o.VIEW = m_currentView;
      }


      private function showFaultEvent(event:FaultEvent):void
      {
        UiMessageBox.show(this, "Błąd ładowania widoku", "Uwaga");
      }

      public override function menuItemClick(event:MenuEvent):void
      {
        var xml:XML = event.item as XML;
        if (xml == null)
          return;

        var stopPropagation:Boolean = true;
        switch (xml.@id.toString())
        {
          case "DEFAULT_VALUE_CONFIG":
            if(grid.dbManager.recordCount < 1)
            {
              UiMessageBox.show(this, "Brak wybranego typu licznika", "Komunikat");
              break;
            }

            var frm:FrmDefaultValue = new FrmDefaultValue();
            frm.meterTypeId = grid.getValue("ID");
            frm.openInContent(UiUtils.getNavigatorContent(this));
            break;
          default:
            stopPropagation = false;
            break;
        }

        if (!stopPropagation)
          super.menuItemClick(event);
      }
